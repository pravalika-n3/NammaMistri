1. PROJECT OVERVIEW

What is Namma-Mistri?

Namma-Mistri is a Construction Assistant Android App designed for rural masons (Mistris) in India. It helps them:

Calculate materials (bricks, cement, sand) needed for walls/rooms
Track labor attendance and wages
Capture site photos to show progress to house owners
Maintain standard rates for local materials
App Features Summary
Feature	Description
Material Calculator	Enter dimensions → Get material list
Labor Diary	Record attendance, advances, balance due
Site Photos	Photo gallery organized by site
Standard Rates	Editable list of local material prices
Multi-Site Support	Manage multiple construction sites

2. DEVELOPMENT ENVIRONMENT SETUP

Step 1: Install Android Studio
text

1. Download Android Studio from: https://developer.android.com/studio
2. Install with default settings
3. SDK Version: API 24 (Android 7.0) minimum, Target API 34
4. Select "Empty Activity" template when creating project

SCREEN 5: MATERIAL CALCULATOR
What it does:
User selects calculation type: Wall or Room
Enters dimensions: Length, Width, Height (in feet)
Selects wall thickness: 4.5 inch (half brick), 9 inch (full brick)
Calculates:
Bricks = Volume × Bricks per cubic foot
Cement bags = Based on mortar ratio
Sand loads = Based on mortar volume
Shows "Calculate" button
Engineering Formulas Used:
text

For Wall:
  Volume (cubic ft) = Length × Height × (Thickness in ft)
  Thickness: 4.5 inch = 0.375 ft, 9 inch = 0.75 ft
  
  Bricks per cubic foot = 13.5 (standard size bricks)
  Total Bricks = Volume × 13.5
  
  Mortar Volume = Volume × 0.3 (30% of wall volume)
  Cement Bags = Mortar Volume × 0.45 / 1.25 (1:6 ratio)
  Sand (cubic ft) = Mortar Volume × 6/7 × 1.54
  Sand Loads = Sand in cubic ft / 100

For Room:
  Calculate all 4 walls and sum up
  Wall1 = Length × Height × Thickness
  Wall2 = Length × Height × Thickness
  Wall3 = Width × Height × Thickness
  Wall4 = Width × Height × Thickness
  Total Volume = 2×(Wall1 + Wall3)

    private int getWorkerDaysPresent(int workerId) {
        SQLiteDatabase db = this.getReadableDatabase();
        // Count full days as 1, half days as 0.5
        Cursor cursorFull = db.rawQuery("SELECT COUNT(*) FROM " + TABLE_ATTENDANCE +
                        " WHERE worker_id=? AND status='present'",
                new String[]{String.valueOf(workerId)});
        int fullDays = 0;
        if (cursorFull.moveToFirst()) fullDays = cursorFull.getInt(0);
        cursorFull.close();

        Cursor cursorHalf = db.rawQuery("SELECT COUNT(*) FROM " + TABLE_ATTENDANCE +
                        " WHERE worker_id=? AND status='half_day'",
                new String[]{String.valueOf(workerId)});
        int halfDays = 0;
        if (cursorHalf.moveToFirst()) halfDays = cursorHalf.getInt(0);
        cursorHalf.close();

        // Return equivalent full days (half days count as 0.5, but we return int, 
        // so handle in calculation)
        return fullDays; // We'll handle half days separately
    }

    public double getWorkerTotalEarning(int workerId) {
        SQLiteDatabase db = this.getReadableDatabase();
        Worker worker = getWorkerById(workerId);
        if (worker == null) return 0;

        Cursor cursorFull = db.rawQuery("SELECT COUNT(*) FROM " + TABLE_ATTENDANCE +
                        " WHERE worker_id=? AND status='present'",
                new String[]{String.valueOf(workerId)});
        int fullDays = 0;
        if (cursorFull.moveToFirst()) fullDays = cursorFull.getInt(0);
        cursorFull.close();

        Cursor cursorHalf = db.rawQuery("SELECT COUNT(*) FROM " + TABLE_ATTENDANCE +
                        " WHERE worker_id=? AND status='half_day'",
                new String[]{String.valueOf(workerId)});
        int halfDays = 0;
        if (cursorHalf.moveToFirst()) halfDays = cursorHalf.getInt(0);
        cursorHalf.close();

        return (fullDays * worker.getDailyWage()) + (halfDays * worker.getDailyWage() * 0.5);
    }

    public void addAdvancePayment(int workerId, double amount) {
        SQLiteDatabase db = this.getWritableDatabase();
        String today = getCurrentDate();

        // Update worker's total advance
        db.execSQL("UPDATE " + TABLE_WORKERS + " SET total_advance = total_advance + ? WHERE id=?",
                new Object[]{amount, workerId});

        // Add attendance record with advance (or update if exists)
        Cursor cursor = db.rawQuery("SELECT id, advance_paid FROM " + TABLE_ATTENDANCE +
                        " WHERE worker_id=? AND date=?",
                new String[]{String.valueOf(workerId), today});

        if (cursor.moveToFirst()) {
            double existingAdvance = cursor.getDouble(1);
            ContentValues cv = new ContentValues();
            cv.put("advance_paid", existingAdvance + amount);
            db.update(TABLE_ATTENDANCE, cv, "worker_id=? AND date=?",
                    new String[]{String.valueOf(workerId), today});
        } else {
            ContentValues cv = new ContentValues();
            cv.put("worker_id", workerId);
            cv.put("date", today);
            cv.put("status", "present");
            cv.put("advance_paid", amount);
            db.insert(TABLE_ATTENDANCE, null, cv);
        }

    }

package com.example.nammamistri;

import android.content.Intent;
import android.os.Bundle;
import android.text.TextUtils;
import android.view.View;
import android.widget.RadioButton;
import android.widget.RadioGroup;
import android.widget.Toast;

import androidx.appcompat.app.AppCompatActivity;

import com.google.android.material.button.MaterialButton;
import com.google.android.material.textfield.TextInputEditText;
import com.google.android.material.textfield.TextInputLayout;

public class MaterialCalculatorActivity extends AppCompatActivity {

    private int siteId;
    private RadioGroup rgCalcType, rgThickness;
    private RadioButton rbWall, rbRoom, rbHalfBrick, rbFullBrick;
    private TextInputEditText etLength, etWidth, etHeight;
    private TextInputLayout tilWidth;
    private MaterialButton btnCalculate;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_material_calculator);

        siteId = getIntent().getIntExtra("site_id", -1);

        if (getSupportActionBar() != null) {
            getSupportActionBar().setTitle("Material Calculator");
            getSupportActionBar().setDisplayHomeAsUpEnabled(true);
        }

        // Initialize views
        rgCalcType = findViewById(R.id.rgCalcType);
        rgThickness = findViewById(R.id.rgThickness);
        rbWall = findViewById(R.id.rbWall);
        rbRoom = findViewById(R.id.rbRoom);
        rbHalfBrick = findViewById(R.id.rbHalfBrick);
        rbFullBrick = findViewById(R.id.rbFullBrick);
        etLength = findViewById(R.id.etLength);
        etWidth = findViewById(R.id.etWidth);
        etHeight = findViewById(R.id.etHeight);
        tilWidth = findViewById(R.id.tilWidth);
        btnCalculate = findViewById(R.id.btnCalculate);

        // Toggle width field visibility based on calc type
        rgCalcType.setOnCheckedChangeListener((group, checkedId) -> {
            if (checkedId == R.id.rbWall) {
                tilWidth.setVisibility(View.GONE);
            } else {
                tilWidth.setVisibility(View.VISIBLE);
            }
        });

        // Initially hide width for wall calculation
        tilWidth.setVisibility(View.GONE);

        // Calculate button
        btnCalculate.setOnClickListener(v -> calculate());
    }

    private void calculate() {
        // Get values
        String lengthStr = etLength.getText().toString().trim();
        String widthStr = etWidth.getText().toString().trim();
        String heightStr = etHeight.getText().toString().trim();

        // Validation
        if (TextUtils.isEmpty(lengthStr)) {
            etLength.setError("Enter length");
            etLength.requestFocus();
            return;
        }
        if (TextUtils.isEmpty(heightStr)) {
            etHeight.setError("Enter height");
            etHeight.requestFocus();
            return;
        }

        boolean isRoom = rbRoom.isChecked();

        if (isRoom && TextUtils.isEmpty(widthStr)) {
            etWidth.setError("Enter width for room");
            etWidth.requestFocus();
            return;
        }

        double length = Double.parseDouble(lengthStr);
        double width = isRoom ? Double.parseDouble(widthStr) : 0;
        double height = Double.parseDouble(heightStr);

        // Wall thickness in feet
        double thickness = rbFullBrick.isChecked() ? 0.75 : 0.375; // 9 inch or 4.5 inch

        // Calculate volume
        double volume;
        if (isRoom) {
            // 4 walls of the room
            // 2 walls of (length × height × thickness) + 2 walls of (width × height × thickness)
            volume = 2 * (length * height * thickness) + 2 * (width * height * thickness);
        } else {
            // Single wall
            volume = length * height * thickness;
        }

        // ===== MATERIAL CALCULATIONS =====

        // Bricks calculation
        // Standard brick size: 9" × 4.5" × 3" with mortar = 9.5" × 5" × 3.5"
        // Bricks per cubic foot ≈ 13.5 (for 9" wall)
        // For 4.5" wall ≈ 13.5 as well (just less volume)
        double bricksPerCuFt = 13.5;
        int bricks = (int) Math.ceil(volume * bricksPerCuFt);

        // Cement calculation
        // Mortar volume = 30% of total wall volume
        // Cement:Sand ratio = 1:6
        // Dry mortar volume factor = 1.54 (33% increase + 21% increase)
        // Cement = (mortar_volume × 1.54 × 1/7) / 0.035 bags (1 bag = 0.035 m³ ≈ 1.25 cft)
        double mortarVolume = volume * 0.30;
        double dryMortarVolume = mortarVolume * 1.54;
        double cementVolume = dryMortarVolume * (1.0 / 7.0);
        int cementBags = (int) Math.ceil(cementVolume / 1.25); // 1 bag = 1.25 cubic feet

        // Sand calculation
        // Sand = (mortar_volume × 1.54 × 6/7) / 100 loads
        double sandVolume = dryMortarVolume * (6.0 / 7.0);
        double sandLoads = sandVolume / 100.0; // 1 load = 100 cubic feet

        // Navigate to result
        Intent intent = new Intent(this, CalculationResultActivity.class);
        intent.putExtra("site_id", siteId);
        intent.putExtra("calc_type", isRoom ? "room" : "wall");
        intent.putExtra("length", length);
        intent.putExtra("width", width);
        intent.putExtra("height", height);
        intent.putExtra("thickness", thickness);
        intent.putExtra("bricks", bricks);
        intent.putExtra("cement_bags", cementBags);
        intent.putExtra("sand_loads", sandLoads);
        startActivity(intent);
    }

    @Override
    public boolean onSupportNavigateUp() {
        finish();
        return true;
    }
}
How it works:

User selects Wall or Room type
For Wall: only length and height needed; For Room: length, width, and height
User selects wall thickness (4.5 inch or 9 inch)
Calculation formulas:
Volume = dimensions × thickness (converted to feet)
Bricks = Volume × 13.5 bricks/cu.ft
Mortar = 30% of wall volume
Cement = Mortar × 1.54 (dry volume) × 1/7 (cement ratio in 1:6 mix)
Sand = Mortar × 1.54 × 6/7 ÷ 100 loads
Results passed via Intent to CalculationResultActivity

        tvBricks = findViewById(R.id.tvBricks);

        tvSand = findViewById(R.id.tvSand);

        btnShare = findViewById(R.id.btnShare);

        // Display results
        displayResults();

        // Save button
        btnSave.setOnClickListener(v -> saveCalculation());

        // Share button
        btnShare.setOnClickListener(v -> shareResult());
    }

    private void displayResults() {
        NumberFormat nf = NumberFormat.getInstance(new Locale("en", "IN"));

        String thicknessInch = thickness == 0.75 ? "9 inch" : "4.5 inch";

        if (calcType.equals("room")) {
            tvDimensions.setText(String.format("📐 Room: %.0f × %.0f × %.0f ft | %s wall",
                    length, width, height, thicknessInch));
        } else {
            tvDimensions.setText(String.format("📐 Wall: %.0f × %.0f ft | %s thick",
                    length, height, thicknessInch));
        }

        tvBricks.setText(nf.format(bricks) + " pieces");
        int bricksWithWastage = (int) Math.ceil(bricks * 1.10); // 10% wastage
        tvBricksWastage.setText("≈ " + nf.format(bricksWithWastage) + " (with 10% wastage)");

        tvCement.setText(cementBags + " bags");

        tvSand.setText(String.format("%.1f loads", sandLoads));
    }

    private void saveCalculation() {
        if (siteId == -1) {
            Toast.makeText(this, "No site associated. Cannot save.", Toast.LENGTH_SHORT).show();
            return;
        }

        Calculation calc = new Calculation();
        calc.setSiteId(siteId);
        calc.setCalcType(calcType);
        calc.setLength(length);
        calc.setWidth(width);
        calc.setHeight(height);
        calc.setWallThickness(thickness);
        calc.setBricksNeeded(bricks);
        calc.setCementBags(cementBags);
        calc.setSandLoads(sandLoads);

        long id = dbHelper.addCalculation(calc);
        if (id > 0) {
            Toast.makeText(this, "Calculation saved successfully!", Toast.LENGTH_SHORT).show();
            btnSave.setEnabled(false);
            btnSave.setText("✅ Saved");
        }
    }

    private void shareResult() {
        String thicknessInch = thickness == 0.75 ? "9 inch" : "4.5 inch";
        NumberFormat nf = NumberFormat.getInstance(new Locale("en", "IN"));
        int bricksWithWastage = (int) Math.ceil(bricks * 1.10);

        StringBuilder sb = new StringBuilder();

            rvPhotos.setVisibility(View.GONE);
        } else {
            tvEmpty.setVisibility(View.GONE);
            rvPhotos.setVisibility(View.VISIBLE);
        }

    private void checkCameraPermission() {
        if (ContextCompat.checkSelfPermission(this, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) {
            ActivityCompat.requestPermissions(this, new String[]{Manifest.permission.CAMERA}, REQUEST_CAMERA_PERMISSION);
        } else {
            takePhoto();
        }
    }

    @Override
    public void onRequestPermissionsResult(int requestCode, @NonNull String[] permissions, @NonNull int[] grantResults) {
        super.onRequestPermissionsResult(requestCode, permissions, grantResults);
        if (requestCode == REQUEST_CAMERA_PERMISSION) {
            if (grantResults.length > 0 && grantResults[0] == PackageManager.PERMISSION_GRANTED) {
                takePhoto();
            } else {
                Toast.makeText(this, "Camera permission is required to take photos", Toast.LENGTH_SHORT).show();
            }
        }
    }

    private void takePhoto() {
        Intent takePictureIntent = new Intent(MediaStore.ACTION_IMAGE_CAPTURE);
        if (takePictureIntent.resolveActivity(getPackageManager()) != null) {
            File photoFile = null;
            try {
                photoFile = createImageFile();
            } catch (IOException ex) {
                Toast.makeText(this, "Error creating photo file", Toast.LENGTH_SHORT).show();
            }

            if (photoFile != null) {
                Uri photoURI = FileProvider.getUriForFile(this,
                        getApplicationContext().getPackageName() + ".fileprovider",
                        photoFile);
                takePictureIntent.putExtra(MediaStore.EXTRA_OUTPUT, photoURI);
                startActivityForResult(takePictureIntent, REQUEST_IMAGE_CAPTURE);
            }
        }
    }

    private File createImageFile() throws IOException {
        String timeStamp = new SimpleDateFormat("yyyyMMdd_HHmmss", Locale.getDefault()).format(new Date());
        String imageFileName = "SITE_" + siteId + "_" + timeStamp;
        File storageDir = getExternalFilesDir(Environment.DIRECTORY_PICTURES);
        File image = File.createTempFile(imageFileName, ".jpg", storageDir);
        currentPhotoPath = image.getAbsolutePath();
        return image;
    }

    @Override
    protected void onActivityResult(int requestCode, int resultCode, @Nullable Intent data) {
        super.onActivityResult(requestCode, resultCode, data);
        if (requestCode == REQUEST_IMAGE_CAPTURE && resultCode == RESULT_OK) {
            // Save photo to database
            SitePhoto photo = new SitePhoto();
            photo.setSiteId(siteId);
            photo.setPhotoPath(currentPhotoPath);
            photo.setDescription("");

            dbHelper.addPhoto(photo);
            Toast.makeText(this, "Photo saved!", Toast.LENGTH_SHORT).show();
            loadPhotos();
        }
    }

    public void onPhotoClick(SitePhoto photo) {
        // Open photo in full screen
        Intent intent = new Intent(Intent.ACTION_VIEW);
        File file = new File(photo.getPhotoPath());
        Uri uri = FileProvider.getUriForFile(this,
                getApplicationContext().getPackageName() + ".fileprovider", file);
        intent.setDataAndType(uri, "image/*");
        intent.addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION);
        startActivity(intent);
    }

        new AlertDialog.Builder(this)
                .setTitle("Delete Photo")
                .setMessage("Are you sure you want to delete this photo?")
                .setPositiveButton("Delete", (dialog, which) -> {
                    // Delete file
                    File file = new File(photo.getPhotoPath());
                    if (file.exists()) file.delete();

                    dbHelper.deletePhoto(photo.getId());
                    loadPhotos();
                })
                .setNegativeButton("Cancel", null)
                .show();
    }

    @Override
    public boolean onSupportNavigateUp() {
        finish();
        return true;

        intent.putExtra("material_name", rate.getMaterialName());
        intent.putExtra("rate", rate.getRate());
        intent.putExtra("unit", rate.getUnit());
        startActivity(intent);
    }

    @Override
    public void onRateLongClick(MaterialRate rate) {
        new AlertDialog.Builder(this)
                .setTitle("Delete Rate")
                .setMessage("Delete '" + rate.getMaterialName() + "'?")
                .setPositiveButton("Delete", (dialog, which) -> {
                    dbHelper.deleteRate(rate.getId());
                    loadRates();
                })
                .setNegativeButton("Cancel", null)
                .show();
    }

    @Override
    public boolean onSupportNavigateUp() {
        finish();
        return true;
    }
}

package com.example.nammamistri;

import android.os.Bundle;
import android.text.TextUtils;
import android.widget.ArrayAdapter;
import android.widget.Spinner;
import android.widget.Toast;

import androidx.appcompat.app.AppCompatActivity;

import com.example.nammamistri.database.DatabaseHelper;
import com.example.nammamistri.models.MaterialRate;
import com.google.android.material.button.MaterialButton;
import com.google.android.material.textfield.TextInputEditText;

public class AddRateActivity extends AppCompatActivity {

    private TextInputEditText etMaterialName, etRate;
    private Spinner spinnerUnit;
    private MaterialButton btnSaveRate;
    private DatabaseHelper dbHelper;

    private int rateId = -1; // -1 means new, otherwise editing
    private String[] units = {"per piece", "per bag", "per load", "per kg", "per sq.ft", "per cu.ft", "per day", "per tanker", "per item"};

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_add_rate);

        if (getSupportActionBar() != null) {
            getSupportActionBar().setTitle("Add/Edit Rate");
            getSupportActionBar().setDisplayHomeAsUpEnabled(true);
        }

        etMaterialName = findViewById(R.id.etMaterialName);
        etRate = findViewById(R.id.etRate);
        spinnerUnit = findViewById(R.id.spinnerUnit);
        btnSaveRate = findViewById(R.id.btnSaveRate);

        dbHelper = new DatabaseHelper(this);

        // Setup unit spinner
        ArrayAdapter<String> adapter = new ArrayAdapter<>(this,
                android.R.layout.simple_spinner_dropdown_item, units);
        spinnerUnit.setAdapter(adapter);

        // Check if editing existing rate
        if (getIntent().hasExtra("rate_id")) {
            rateId = getIntent().getIntExtra("rate_id", -1);
            String materialName = getIntent().getStringExtra("material_name");
            double rate = getIntent().getDoubleExtra("rate", 0);
            String unit = getIntent().getStringExtra("unit");

            etMaterialName.setText(materialName);
            etRate.setText(String.valueOf(rate));

            // Set spinner to correct unit
            for (int i = 0; i < units.length; i++) {
                if (units[i].equals(unit)) {
                    spinnerUnit.setSelection(i);
                    break;
                }
            }

            btnSaveRate.setText("💾 UPDATE RATE");
        }

        btnSaveRate.setOnClickListener(v -> saveRate());
    }

    private void saveRate() {
        String name = etMaterialName.getText().toString().trim();
        String rateStr = etRate.getText().toString().trim();

        if (TextUtils.isEmpty(name)) {
            etMaterialName.setError("Material name required");
            etMaterialName.requestFocus();
            return;
        }
        if (TextUtils.isEmpty(rateStr)) {
            etRate.setError("Rate is required");
            etRate.requestFocus();
            return;
        }

        double rate = Double.parseDouble(rateStr);
        String unit = spinnerUnit.getSelectedItem().toString();

        MaterialRate materialRate = new MaterialRate();
        materialRate.setMaterialName(name);
        materialRate.setRate(rate);
        materialRate.setUnit(unit);

        if (rateId != -1) {
            // Update existing
            materialRate.setId(rateId);
            dbHelper.updateRate(materialRate);
            Toast.makeText(this, "Rate updated!", Toast.LENGTH_SHORT).show();
        } else {
            // Add new
            dbHelper.addRate(materialRate);
            Toast.makeText(this, "Rate added!", Toast.LENGTH_SHORT).show();
        }

        finish();
    }

    @Override
    public boolean onSupportNavigateUp() {
        finish();
        return true;
    }
}
